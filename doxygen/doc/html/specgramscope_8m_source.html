1 function varargout = specgramscope(varargin)
2 % SPECGRAMSCOPE Live updating spectrogram display
3 %
4 % STEP 1: Initialize the scope
5 % SPECGRAMSCOPE(FS,NFFT) initializes a spectrogram scope in the current axes.
6 % This spectrogram scope will compute and displays the NFFT-point FFT of a vector
7 % signal with sample rate FS Hz. It will show the last 10 FFTs
8 %
9 % STEP 2: Update the scope
10 % SPECGRAMSCOPE(S) updates the spectrogram scope in the current axes with the
11 % FFT of vector S. The scope should first be initialized as above with
12 % sample rate and FFT length. If not, the sample rate will be 1 Hz and the FFT
13 % length will be the length of S. Differences between the length of S and
14 % the specified FFT length are handled the same as MATLAB's built-in FFT
15 % function (i.e., zero-padding or truncation, as appropriate).
16 %
17 % SPECGRAMSCOPE(FS,NFFT,NTRACES) initializes a spectrogram scope in the
18 % current axes with NTRACES traces. This is how many records to keep in
19 % time domain. Default = 10
20 %
21 % SPECGRAMSCOPE(HAX, ...) defines the scope in specified axes HAX instead of GCA. i.e.,
22 % SPECGRAMSCOPE(HAX,FS,NFFT) initializes axes HAX as a spectrogram scope, and
23 % SPECGRAMSCOPE(HAX,S) updates axes HAX with vector S.
24 %
25 % HAX = SPECGRAMSCOPE(...) returns a handle to the axes initialized by the
26 % spectrogram scope. This is useful if you allow SPECGRAMSCOPE to create an
27 % axes for you, and want to be able to easily reference the axes for
28 % updates. The surface created by SPECGRAMSCOPE all have the tag
29 % 'SpecgramScope'. If you would like to manually modify the properties of
30 % these lines, their handles can be found by:
31 %
32 % HAX = SPECGRAMSCOPE(...);
33 % HSurf = findobj(HAX,'Tag','SpecgramScope');
34 %
35 % Example
36 % %% Initialize data
37 % Fs = 16384;
38 % Nfft = 2048;
39 % t = (0:1:Nfft-1)'/Fs;
40 % fo = logspace(3.5,3.7); % Range of fundamental frequencies
41 % s1 = sin(2*pi*t*fo) + .1*rand(Nfft,length(fo));
42 %
43 %
44 % %% Initialize scope
45 % specgramscope(Fs,Nfft,30);
46 % view([103 30])
47 %
48 % %% Update scope
49 % for ii = 1:length(fo)
50 % specgramscope(s1(:,ii));
51 % drawnow;pause(.01);
52 % end;

59 %% Parse input arguments
60 % Decision tree:
61 % + Initialize or update?
62 % o If update -> OK
63 % o If initialize -> Axes specified, or use GCA?
64 
65 error(nargchk(1,4,nargin))
66 NTracesDefault = 10;
67 
68 %% Initialize or update?
69 % If first or second input argument is not a scalar, it must be data - i.e. we are
70 % updating
71 
72 if prod(size(varargin{1})) > 1 | prod(size(varargin{2})) > 1 % Update
73  action = 'update';
74 
75  if nargin==1 % Use current axes
76  hAxes = gca;
77  data = varargin{1};
78  else
79  hAxes = varargin{1}; % Axes was specified
80  data = varargin{2};
81  end;
82 
83  % If the user has not initialized this scope, do it for them
84  parms = getappdata(hAxes,'SpecgramScopeParameters');
85 
86  % Ensure that scope has been initialized
87  if isempty(parms)
88  % Use default values
89  Fs = 1;
90  data = rowmajor(data);
91  Nfft = length(data);
92  NTraces = NTracesDefault; % Number of time histories
93  feval(mfilename,hAxes,Fs,Nfft,NTraces); % This recursive call will initialize the scope
94  % Get the new parameter structure
95  parms = getappdata(hAxes,'SpecgramScopeParameters');
96  end;
97 
98 
99 
100 else % Initialize
101  action = 'init';
102 
103  if ~isaxes(varargin{1}) % Easy mode, no handle passed in
104  % Use current axes
105  hAxes = gca;
106  Fs = varargin{1};
107  Nfft = varargin{2};
108  if nargin==3
109  NTraces = varargin{3};
110  else
111  NTraces = NTracesDefault;
112  end;
113 
114  else % Expert mode, passed handle in
115  hAxes = varargin{1};
116  Fs = varargin{2};
117  Nfft = varargin{3};
118  if nargin==4
119  NTraces = varargin{4};
120  else
121  NTraces = NTracesDefault;
122  end;
123  end;
124 end;

126 %% Dole out the work
127 %
128 switch action
129  case 'init' % Initialize
130 
131  % Build structure to internally pass information
132  parms.Fs = Fs; % Sample Rate
133  parms.NTraces = NTraces; % Number of records in time
134  parms.hAxes = hAxes; % Handle to axes
135  parms.Nfft = Nfft; % FFT Block size
136 
137  % Store parameter structure
138  setappdata(hAxes,'SpecgramScopeParameters',parms);
139 
140  localInitScope(parms) % Initialize scope
141 
142  case 'update' % Update
143  parms = getappdata(hAxes,'SpecgramScopeParameters');
144 
145  % Error checking
146  % Ensure that scope has been initialized. This shouldn't slip
147  % through to here.
148  if isempty(parms)
149  error(['The spectrogram scope must first be initialized ' ...
150  'with the sample rate: specgramscope(hAxes,Fs)']);
151  end;
152 
153  % Force data to be in columns. Allow for multiple columns. This will
154  % error if data actually has more channels than samples.
155  data = rowmajor(data);
156 
157  % Check that the number of columns corresponds to the number of lines
158  nc = size(data,2); % Number of columns
159  if nc ~= 1
160  error(['spectrogram scope requires a single column vector of data']);
161  end;
162 
163  localUpdateScope(data,parms) % Update the scope
164 end;
165 
166 % Return appropriate output argument
167 if nargout
168  varargout{1} = parms.hAxes;
169 end;

174 function localInitScope(parms)
175 
176 % Set axes
177 f = (0:parms.Nfft/2-1)*parms.Fs/parms.Nfft;
178 f = f(:);
179 
180 % Add surface
181 [X,Y] = meshgrid(-parms.NTraces+1:0,f);
182 parms.hSurf = surf(parms.hAxes,X,Y,NaN*ones(length(f),parms.NTraces), ...
183  'Tag','SpecgramScope');
184 
185 set(parms.hAxes, ...
186  'XLim',[-parms.NTraces+1 0], ...
187  'YLim',[0 f(end)]);%, ...
188 % 'YDir','reverse'); % Reverse frequency axes direction
189 shading(parms.hAxes,'interp');
190 
191 setappdata(parms.hAxes,'SpecgramScopeParameters',parms);
192 
193 %% Get handle to the figure
194 % Turn doublebuffer on to eliminate flickering
195 hFig = get(parms.hAxes,'Parent');
196 
197 % In R14, it's possible that hFig would return a handle to a panel, not a
198 % figure
199 if ~strcmp(get(hFig,'Type'),'figure')
200  hFig = get(hFig,'Parent');
201 end;
202 
203 %%
204 % Label the plot.
205 % There's a bug in R13 when creating xlabel and ylabel with direct

222 function localUpdateScope(data,parms)
223 
224 [f,mag] = localfft(data,parms);
225 
226 % Dynamically modify Magnitude axis as we go. Expand, but don't shrink.
227 maxM=max(mag(:));
228 minM=min(mag(:));
229 yax2=get(parms.hAxes,'YLim');
230 if minM<yax2(1),
231  yax2(1)=minM;
232 end
233 if maxM>yax2(2),
234  yax2(2)=maxM;
235 end
236 set(parms.hAxes,'YLim',yax2)

239 % Update the plot
240 hSurf = parms.hSurf;

243 set(hSurf,'ZData',zd,'CData',zd)

252 % ***********************************************************************
253 % Calculate the fft of the data.
254 function [f, mag] = localfft(data,parms)
255 
256 % Calculate the fft of the data.
257 xfft = 2/parms.Nfft*fft(data,parms.Nfft);
258 
259 % Avoid taking the log of 0.
260 xfft(xfft == 0) = 1e-17;
261 
262 % Compute magnitude, dB
263 mag = 20*log10(abs(xfft(1:parms.Nfft/2,:)));
264 
265 f = (0:length(mag)-1)*parms.Fs/parms.Nfft;
266 f = f(:);
