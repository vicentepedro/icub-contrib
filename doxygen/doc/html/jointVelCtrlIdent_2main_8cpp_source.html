319  void goToStartingPos()
320  {
321  ivel->stop(joint);
322  imod->setControlMode(joint,VOCAB_CM_POSITION);
323  ipos->setRefSpeed(joint,20.0);
324  ipos->positionMove(joint,fb0);
325 
326  bool done=false;
327  while (!done)
328  {
329  double fb;
330  ienc->getEncoder(joint,&fb);
331  done=fabs(fb0-fb)<0.5;
332  Time::delay(0.1);
333  }
334 
335  ipos->stop(joint);
336  imod->setControlMode(joint,VOCAB_CM_VELOCITY);
337  ivel->setRefAcceleration(joint,1e9);
338  velEst.reset();
339  }

341  /***********************************************************/
342  void run()
343  {
344  Bottle &out=port.prepare();
345  out.clear();
346 
347  double t=Time::now()-t0;
348  if (t>T)
349  {
350  goToStartingPos();
351  t0=Time::now();
352  t=0.0;
353  cnt++;
354  }
355 
356  int mode=cnt%3;
357  double fb; ienc->getEncoder(joint,&fb);
358  double ref; ipid->getReference(joint,&ref);
359  Vector pwm=getOutput();
360  Vector vel(1), e(1); e[0]=fb1-fb;
361 
362  if (mode==0)
363  vel[0]=gain*sin(2.0*M_PI*(fmin+((fmax-fmin)/T)*t)*t);
364  else if (mode==1)
365  vel=ctrl1->computeCmd(Ttraj,e);
366  else
367  vel=ctrl2->computeCmd(Ttraj,e);
368 
369  ivel->velocityMove(joint,vel[0]);
370 
371  AWPolyElement el;
372  el.data.resize(1,fb);
373  el.time=Time::now();
374  Vector dfb=velEst.estimate(el);
375 
376  out.addInt(cnt);
377  out.addDouble(vel[0]);
378  out.addDouble(ref);
379  for (size_t i=0; i<pwm.length(); i++)
380  out.addDouble(pwm[i]);
381  out.addDouble(fb);
382  out.addDouble(dfb[0]);
383 
384  port.write();
385  }
