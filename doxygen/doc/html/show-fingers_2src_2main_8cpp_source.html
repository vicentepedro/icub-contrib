43 /************************************************************************/
44 class FingerModule : public RFModule
45 {
46 protected:
47  iCubFinger finger[3];
48  int camSel,nEncs;
49 
50  PolyDriver drvArm,drvCart,drvGaze;
51  IEncoders *iencs;
52  ICartesianControl *iarm;
53  IGazeControl *igaze;
54 
55  BufferedPort<ImageOf<PixelRgb> > imgInPort,imgOutPort;
56 
57 public:
58  /************************************************************************/
59  bool configure(ResourceFinder &rf)
60  {
61  string robot=rf.check("robot",Value("icub")).asString();
62  string arm=rf.check("arm",Value("left")).asString();
63  string eye=rf.check("eye",Value("left")).asString();
64 
65  if ((arm!="left") && (arm!="right"))
66  {
67  yError()<<"Invalid arm requested";
68  return false;
69  }
70 
71  if ((eye!="left") && (eye!="right"))
72  {
73  yError()<<"Invalid eye requested";
74  return false;
75  }
76 
77  // open drivers
78  Property optionArm("(device remote_controlboard)");
79  optionArm.put("remote",("/"+robot+"/"+arm+"_arm").c_str());
80  optionArm.put("local","/show-fingers/joints");
81  if (!drvArm.open(optionArm))
82  {
83  yError()<<"Joints arm controller not available";
84  terminate();
85  return false;
86  }
87 
88  Property optionCart("(device cartesiancontrollerclient)");
89  optionCart.put("remote",("/"+robot+"/cartesianController/"+arm+"_arm").c_str());
90  optionCart.put("local","/show-fingers/cartesian");
91  if (!drvCart.open(optionCart))
92  {
93  yError()<<"Cartesian arm controller not available";
94  terminate();
95  return false;
96  }
97 
98  Property optionGaze("(device gazecontrollerclient)");
99  optionGaze.put("remote","/iKinGazeCtrl");
100  optionGaze.put("local","/show-fingers/gaze");
101  if (!drvGaze.open(optionGaze))
102  {
103  yError()<<"Gaze controller not available";
104  terminate();
105  return false;
106  }
107 
108  IControlLimits *ilim;
109  drvArm.view(iencs);
110  drvArm.view(ilim);
111  drvCart.view(iarm);
112  drvGaze.view(igaze);
113  iencs->getAxes(&nEncs);
114 
115  finger[0]=iCubFinger(arm+"_thumb");
116  finger[1]=iCubFinger(arm+"_index");
117  finger[2]=iCubFinger(arm+"_middle");
118 
119  deque<IControlLimits*> lim;
120  lim.push_back(ilim);
121  for (int i=0; i<3; i++)
122  finger[i].alignJointsBounds(lim);
123 
124  yInfo()<<"Using arm="<<arm<<" and eye="<<eye;
125 
126  imgInPort.open("/show-fingers/img:i");
127  imgOutPort.open("/show-fingers/img:o");
128 
129  camSel=(eye=="left")?0:1;
130  return true;
131  }
132 
133  /************************************************************************/
134  double getPeriod()
135  {
136  return 0.0;
137  }

139  /************************************************************************/
140  bool updateModule()
141  {
142  ImageOf<PixelRgb> *imgIn=imgInPort.read(true);
143  if (imgIn==NULL)
144  return false;
145 
146  ImageOf<PixelRgb> &imgOut=imgOutPort.prepare();
147  imgOut=*imgIn;
148 
149  cv::Mat img=cv::cvarrToMat(imgOut.getIplImage());
150 
151  Vector xa,oa;
152  iarm->getPose(xa,oa);
153 
154  Matrix Ha=axis2dcm(oa);
155  xa.push_back(1.0);
156  Ha.setCol(3,xa);
157 
158  Vector pc;
159  igaze->get2DPixel(camSel,xa,pc);
160 
161  cv::Point point_c((int)pc[0],(int)pc[1]);
162  cv::circle(img,point_c,4,cv::Scalar(0,255,0),4);
163 
164  Vector encs(nEncs),joints;
165  iencs->getEncoders(encs.data());
166  for (int i=0; i<3; i++)
167  {
168  finger[i].getChainJoints(encs,joints);
169  finger[i].setAng(CTRL_DEG2RAD*joints);
170  }
171 
172  for (int fng=0; fng<3; fng++)
173  {
174  deque<cv::Point> point_f;
175  for (size_t i=0; i<finger[fng].getN(); i++)
176  {
177  Vector fc;
178  igaze->get2DPixel(camSel,Ha*finger[fng].getH(i,true).getCol(3),fc);
179  point_f.push_front(cv::Point((int)fc[0],(int)fc[1]));
180  cv::circle(img,point_f.front(),3,cv::Scalar(0,0,255),4);
181 
182  if (i>0)
183  {
184  cv::line(img,point_f.front(),point_f.back(),cv::Scalar(255,255,255),2);
185  point_f.pop_back();
186  }
187  else
188  cv::line(img,point_c,point_f.front(),cv::Scalar(255,0,0),2);
189  }
190  }
191 
192  imgOutPort.writeStrict();
193  return true;
194  }

210

213  }

227  }
