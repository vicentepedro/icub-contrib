1 function varargout = stripchart(varargin)
2 % STRIPCHART Create a stripchart for real-time data display
3 %
4 % STRIPCHART makes it fairly easy to include a strip chart in your
5 % real-time data acquisition and analysis application. You feed
6 % stripchart your data, and it updates the display. It takes
7 % 2 steps to use STRIPCHART. First, you initialize the stripchart with basic
8 % information needed for the FFT (sample rate, axis width, number of traces).
9 % After that, all you need to do is pass your data to the scope to update.
10 %
11 % This documentation starts with the simplest syntax for the two steps,
12 % then provides a few more advanced options.
13 %
14 % STEP 1: Initialize the stripchart
15 % STRIPCHART(FS,AXESWIDTH) initializes a stripchart in the current axes.
16 % This stripchart will display incoming data with a sample rate of FS Hz.
17 % The stripchart will display the most recent AXESWIDTH seconds of data.
18 %
19 % STEP 2: Update the scope
20 % STRIPCHART(S) updates the stripchart in the current axes with the
21 % data in vector S. The scope should first be initialized as above with
22 % sample rate and axes width. If not, the sample rate will be 1 Hz and the
23 % axes width will match the duration of S.
24 %
25 % STRIPCHART(FS,AXESWIDTH,NTRACES) initializes a stripchart in the
26 % current axes with NTRACES traces. A trace is a single line on the scope;
27 % typically one will display one trace per channel of data. The default
28 % for NTRACES is 1. To update a stripchart with multiple traces,
29 % STRIPCHART(S) must specify a matrix S with shorter dimension length =
30 % NTRACES.
31 %
32 % STRIPCHART(HAX, ...) defines the scope in specified axes HAX instead of GCA. i.e.,
33 % STRIPCHART(HAX,FS,AXESWIDTH) initializes axes HAX as a stripchart, and
34 % STRIPCHART(HAX,S) updates axes HAX with vector S.
35 %
36 % HAX = STRIPCHART(...) returns a handle to the axes initialized by the
37 % stripchart. This is useful if you allow STRIPCHART to create an
38 % axes for you, and want to be able to easily reference the axes for
39 % updates. The lines created by STRIPCHART all have the tag
40 % 'StripChart'. If you would like to manually modify the properties of
41 % these lines, their handles can be found by:
42 %
43 % HAX = STRIPCHART(...);
44 % HLINE = findobj(HAX,'Tag','StripChart');
45 %
46 % Example
47 % %% Create data
48 % Fs = 1000; % Sample rate
49 % Ns = Fs*3; % Make 5 seconds worth of data
50 % t = (0:1:Ns-1)'/Fs;
51 % A = sqrt(t);
52 % A(1:Ns/2) = A(1:Ns/2);
53 % A(end:-1:Ns/2+1) = A(1:Ns/2);
54 % s = A.*sin(2*pi*t*1);
55 %
56 % %% Initialize scope
57 % clf
58 % AxesWidth = 2; % Axes Width (s)
59 % stripchart(Fs,AxesWidth);
60 %
61 % %% Update scope
62 % N = 50;
63 % ind = 1:N:Ns;
64 %
65 % for ii = ind
66 % stripchart(s(ii:ii+N-1,:));
67 % drawnow;pause(.05);
68 % end;

75 %% Parse input arguments
76 % Decision tree:
77 % + Initialize or update?
78 % o If update -> OK
79 % o If initialize -> Axes specified, or use GCA?
80 
81 error(nargchk(1,4,nargin))

83 %% Initialize or update?
84 % If first or second input argument is not a scalar, it must be data - i.e. we are
85 % updating
86 
87 if numel(varargin{1}) > 1 || numel(varargin{2}) > 1 % Update
88  action = 'update';
89 
90  if nargin==1 % Use current axes
91  hAxes = gca;
92  data = varargin{1};
93  else
94  hAxes = varargin{1}; % Axes was specified
95  data = varargin{2};
96  end;
97 
98  % If the user has not initialized this scope, do it for them
99  parms = getappdata(hAxes,'StripChartParameters');
100 
101  % Ensure that scope has been initialized
102  if isempty(parms)
103  % Use default values
104  Fs = 1;
105  data = rowmajor(data);
106  [Ns,NTraces] = size(data);
107  AxesWidth = Ns*Fs;
108  feval(mfilename,hAxes,Fs,AxesWidth,NTraces); % This recursive call will initialize the scope
109  % Get the new parameter structure
110  parms = getappdata(hAxes,'StripChartParameters');
111  end;
112 
113 
114 
115 else % Initialize
116  action = 'init';
117 
118  if ~isaxes(varargin{1}) % Easy mode, no handle passed in
119  % Use current axes
120  hAxes = gca;
121  Fs = varargin{1};
122  AxesWidth = varargin{2};
123  Ns = AxesWidth*Fs; % Number of samples across axes
124  if nargin==3
125  NTraces = varargin{3};
126  else
127  NTraces = 1;
128  end;
129 
130  else % Expert mode, passed handle in
131  hAxes = varargin{1};
132  Fs = varargin{2};
133  AxesWidth = varargin{3};
134  Ns = AxesWidth*Fs; % Number of samples across axes
135  if nargin==4
136  NTraces = varargin{4};
137  else
138  NTraces = 1;
139  end;
140  end;
141 end;

144 %% Dole out the work
145 %
146 switch action
147  case 'init' % Initialize
148 
149  % Build structure to internally pass information
150  parms.Fs = Fs; % Sample Rate
151  parms.NTraces = NTraces; % Number of lines in plot
152  parms.hAxes = hAxes; % Handle to axes
153  parms.Ns = Ns; % Number of samples across axes
154  parms.AxesWidth = AxesWidth; % Requested Axes Width (s)
155 
156  % Store parameter structure
157  setappdata(hAxes,'StripChartParameters',parms);
158 
159  localInitScope(parms) % Initialize scope
160 
161  case 'update' % Update
162  parms = getappdata(hAxes,'StripChartParameters');
163 
164  % Error checking
165  % Ensure that scope has been initialized. This shouldn't slip
166  % through to here.
167  if isempty(parms)
168  error(['The spectrum scope must first be initialized ' ...
169  'with the sample rate: stripchart(hAxes,Fs)']);
170  end;
171 
172  % Force data to be in columns. Allow for multiple columns. This will
173  % error if data actually has more channels than samples.
174  data = rowmajor(data);
175 
176  % Check that the number of columns corresponds to the number of lines
177  nc = size(data,2); % Number of columns
178  if nc ~= parms.NTraces
179  error(['Size mismatch. You initialized stripchart with ' num2str(parms.NTraces) ...
180  ' lines, but just passed in ' num2str(nc) ' channels of data. These' ...
181  ' numbers must be the same.']);
182  end;
183 
184  localUpdateScope(data,parms) % Update the scope
185 end;
186 
187 % Return appropriate output argument
188 if nargout
189  varargout{1} = parms.hAxes;
190 end;

194 % Initialize the Scope
195 function localInitScope(parms)
196 
197 % Set axes
198 t = (0:1:parms.Ns-1)'/parms.Fs;
199 
200 % Add line(s)
201 parms.hLine = plot(t,NaN*ones(length(t),parms.NTraces), ...
202  'Tag','StripChart', ...
203  'Parent',parms.hAxes);
204 set(parms.hAxes,'XLim',[0 parms.AxesWidth]);
205 setappdata(parms.hAxes,'StripChartParameters',parms);
206 
207 %% Get handle to the figure
208 % Turn doublebuffer on to eliminate flickering
209 hFig = get(parms.hAxes,'Parent');
210 
211 % In R14, it's possible that hFig would return a handle to a panel, not a
212 % figure
213 if ~strcmp(get(hFig,'Type'),'figure')
214  hFig = get(hFig,'Parent');
215 end;
216 
217 %%
218 % Format the axes
219 % maintain X tick spacing
220 Ticks = get(parms.hAxes,'XTick');
221 dX = mean(diff(Ticks));
222 
223 % enforce right justification for grid lines & remove X tick labels
224 Range = get(parms.hAxes,'XLim');
225 if Range(end)~=Ticks(end)
226  Ticks = fliplr(Range(end):-dX:Range(1));
227  set(parms.hAxes,'XTick',Ticks)
228 end
229 set(parms.hAxes,'XTickLabel',[],'XTickMode','Manual','XGrid','On','YGrid','On')
230 
231 %%
232 % Label the plot.
233 % There's a bug in R13 when creating xlabel and ylabel with direct
234 % parenting - the alignment gets all messed up. Instead, make hAx current
235 % axes
236 ca = gca;
237 set(hFig,'CurrentAxes',parms.hAxes);
238 xlabel(sprintf('%g%s/div',dX,'s'))
239 ylabel('Amplitude');
240 set(hFig,'CurrentAxes',ca);
241 
242 axis manual
243

247 set(hFig,'DoubleBuffer','on');

250 % Update the plot.
251 function localUpdateScope(data,parms)
252 % Callback function to update stripchart with new data
253 
254 % Dynamically modify Magnitude axis as we go. Expand, but don't shrink.
255 maxM=max(data(:));
256 minM=min(data(:));
257 yax2=get(parms.hAxes,'YLim');
258 if minM<yax2(1),
259  yax2(1)=minM;
260 end
261 if maxM>yax2(2),
262  yax2(2)=maxM;
263 end
264 set(parms.hAxes,'YLim',yax2)
265 
266 
267 hLine = parms.hLine;
268 
269 [newPts,NLines] = size(data);
270 yData = get(hLine,'YData'); % old data
271 
272 if NLines==1 %Special case for one line only
273  yData(1:end-newPts) = yData(newPts+1:end); % shift old data left
274  yData(end-newPts+1:end) = data; % new data goes on right
275  set(hLine,'YData',yData) % update plot
276 else
277  for ii=1:NLines
278  yData{ii}(1:end-newPts) = yData{ii}(newPts+1:end); % shift old data left
279  yData{ii}(end-newPts+1:end) = data(:,ii); % new data goes on right
280  end;
281  set(hLine,{'YData'},yData) % update plot
282 end;
