182  bool respond(const Bottle &command, Bottle &reply)
183  {
184  int on=Vocab::encode("on");
185  int off=Vocab::encode("off");
186  int ack=Vocab::encode("ack");
187  int nack=Vocab::encode("nack");
188 
189  if (command.size()>0)
190  {
191  switch (command.get(0).asVocab())
192  {
193  //-----------------
194  case VOCAB4('l','o','a','d'):
195  {
196  if (command.size()>=2)
197  {
198  if (command.get(1).isString())
199  {
200  mutex.lock();
201  string model=rf->findFileByName(command.get(1).asString()).c_str();
202  tld.readFromFile(model.c_str());
203  firstInitTLD=true;
204  mutex.unlock();
205 
206  reply.addVocab(ack);
207  reply.addString(model.c_str());
208  }
209  else if (command.size()>=5)
210  {
211  mutex.lock();
212  boundingBox.x=command.get(1).asInt();
213  boundingBox.y=command.get(2).asInt();
214  boundingBox.width=command.get(3).asInt()-boundingBox.x;
215  boundingBox.height=command.get(4).asInt()-boundingBox.y;
216  loadBoundingBox=true;
217  mutex.unlock();
218 
219  reply.addVocab(ack);
220  }
221  else
222  reply.addVocab(nack);
223  }
224  else
225  reply.addVocab(nack);
226 
227  return true;
228  }
229 
230  //-----------------
231  case VOCAB4('s','a','v','e'):
232  {
233  if (command.size()>=2)
234  {
235  mutex.lock();
236  string model=rf->getHomeContextPath().c_str();
237  model+="/";
238  model+=command.get(1).asString().c_str();
239  tld.writeToFile(model.c_str());
240  mutex.unlock();
241 
242  reply.addVocab(ack);
243  reply.addString(model.c_str());
244  }
245  else
246  reply.addVocab(nack);
247 
248  return true;
249  }
250 
251  //-----------------
252  case VOCAB4('s','t','a','r'):
253  {
254  if (firstInitTLD)
255  {
256  doTLD=true;
257  reply.addVocab(ack);
258  }
259  else
260  reply.addVocab(nack);
261 
262  return true;
263  }
264 
265  //-----------------
266  case VOCAB4('s','t','o','p'):
267  {
268  doTLD=false;
269  reply.addVocab(ack);
270  return true;
271  }
272 
273  //-----------------
274  case VOCAB4('c','l','e','a'):
275  {
276  mutex.lock();
277  tld.release();
278  mutex.unlock();
279 
280  reply.addVocab(ack);
281  return true;
282  }
283 
284  //-----------------
285  case VOCAB3('s','e','t'):
286  {
287  if (command.size()>=3)
288  {
289  int subcmd=command.get(1).asVocab();
290  int mode=command.get(2).asVocab();
291  switch (subcmd)
292  {
293  //-----------------
294  case VOCAB4('l','e','a','r'):
295  {
296  if ((mode==on) || (mode==off))
297  {
298  mutex.lock();
299  tld.learningEnabled=(mode==on);
300  mutex.unlock();
301  reply.addVocab(ack);
302  }
303  else
304  reply.addVocab(nack);
305 
306  break;
307  }
308 
309  //-----------------
310  case VOCAB4('a','l','t','e'):
311  {
312  if ((mode==on) || (mode==off))
313  {
314  mutex.lock();
315  tld.alternating=(mode==on);
316  mutex.unlock();
317  reply.addVocab(ack);
318  }
319  else
320  reply.addVocab(nack);
321 
322  break;
323  }
324 
325  //-----------------
326  default:
327  reply.addVocab(nack);
328  }
329  }
330  else
331  reply.addVocab(nack);
332 
333  return true;
334  }
335 
336  //-----------------
337  case VOCAB3('g','e','t'):
338  {
339  if (command.size()>=2)
340  {
341  int subcmd=command.get(1).asVocab();
342  switch (subcmd)
343  {
344  //-----------------
345  case VOCAB4('l','e','a','r'):
346  {
347  reply.addVocab(ack);
348  reply.addVocab(tld.learningEnabled?on:off);
349  break;
350  }
351 
352  //-----------------
353  case VOCAB4('a','l','t','e'):
354  {
355  reply.addVocab(ack);
356  reply.addVocab(tld.alternating?on:off);
357  break;
358  }
359 
360  //-----------------
361  default:
362  reply.addVocab(nack);
363  }
364  }
365  else
366  reply.addVocab(nack);
367 
368  return true;
369  }
370 
371  //-----------------
372  default:
373  return RFModule::respond(command,reply);
374  }
375  }
376 
377  reply.addVocab(nack);
378  return true;
379  }

387  /************************************************************************/
388  bool updateModule()
389  {
390  ImageOf<PixelBgr> *img=imgInPort.read();
391  if (img==NULL)
392  return true;
393 
394  mutex.lock();
395 
396  if (initDetectorCascade)
397  {
398  ImageOf<PixelMono> imgMono;
399  imgMono.resize(img->width(),img->height());
400  tld.detectorCascade->imgWidth=imgMono.width();
401  tld.detectorCascade->imgHeight=imgMono.height();
402  tld.detectorCascade->imgWidthStep=((IplImage*)imgMono.getIplImage())->widthStep;
403  initDetectorCascade=false;
404  }
405 
406  bool skipProcessing=false;
407  if (loadBoundingBox)
408  {
409  ImageOf<PixelMono> imgMono;
410  imgMono.resize(img->width(),img->height());
411  cvCvtColor(img->getIplImage(),imgMono.getIplImage(),CV_BGR2GRAY);
412 
413  tld.selectObject((IplImage*)imgMono.getIplImage(),&boundingBox);
414  loadBoundingBox=false;
415  skipProcessing=true;
416  firstInitTLD=true;
417  }
418 
419  if (doTLD && !skipProcessing)
420  {
421  tld.processImage((IplImage*)img->getIplImage());
422 
423  if (tld.currBB!=NULL)
424  {
425  CvPoint tl,br;
426  tl.x=tld.currBB->x;
427  tl.y=tld.currBB->y;
428  br.x=tl.x+tld.currBB->width;
429  br.y=tl.y+tld.currBB->height;
430 
431  if (dataOutPort.getOutputCount()>0)
432  {
433  Bottle data;
434  data.addInt(tl.x);
435  data.addInt(tl.y);
436  data.addInt(br.x);
437  data.addInt(br.y);
438  data.addDouble(tld.currConf);
439  dataOutPort.write(data);
440  }
441 
442  if (imgOutPort.getOutputCount()>0)
443  {
444  cvRectangle(img->getIplImage(),tl,br,(tld.currConf>=0.5)?cvScalar(255,0,0):cvScalar(0,255,255),2);
445  imgOutPort.write(*img);
446  }
447  }
448  }
449 
450  mutex.unlock();
451 
452  return true;
453  }
